<script>
let cart = [];

// LIVRES
const books = [
    {name:"Business Rapide", price:6000, category:"business", img:"https://via.placeholder.com/150"},
    {name:"Argent avec le téléphone", price:6500, category:"business", img:"https://via.placeholder.com/150"},
    {name:"Secrets Marketing", price:5000, category:"business", img:"https://via.placeholder.com/150"},
    {name:"Mentalité Gagnante", price:4500, category:"motivation", img:"https://via.placeholder.com/150"},
    {name:"Le Pouvoir Caché", price:5000, category:"motivation", img:"https://via.placeholder.com/150"},
    {name:"Réussir en Afrique", price:5500, category:"motivation", img:"https://via.placeholder.com/150"}
];

// AFFICHAGE
function displayBooks(list, containerId){
    let container = document.getElementById(containerId);
    container.innerHTML = "";

    list.forEach(book=>{
        container.innerHTML += `
        <div class="book">
            <img src="${book.img}">
            <h4>${book.name}</h4>
            <p>${book.price} FCFA</p>
            <button onclick="addToCart('${book.name}', ${book.price})">Ajouter</button>
            <button onclick="buyNow('${book.name}', ${book.price})">Acheter</button>
        </div>
        `;
    });
}

// INITIAL
displayBooks(books, "allBooks");

// PANIER
function addToCart(name, price){
    cart.push({name, price});
    document.getElementById("count").innerText = cart.length;
}

// WHATSAPP
function buyNow(name, price){
    let message = `Bonjour, je veux acheter ${name} à ${price} FCFA`;
    let phone = "22665623164";
    window.open(`https://wa.me/${phone}?text=${encodeURIComponent(message)}`);
}

// RECHERCHE
function searchBook(value){
    value = value.toLowerCase();

    if(value === ""){
        document.getElementById("home").style.display = "block";
        document.getElementById("searchResult").style.display = "none";
        return;
    }

    let found = books.find(book => book.name.toLowerCase().includes(value));

    if(found){
        document.getElementById("home").style.display = "none";
        document.getElementById("searchResult").style.display = "block";

        displayBooks([found], "resultBooks");

        let related = books.filter(b => b.category === found.category && b.name !== found.name);
        displayBooks(related, "relatedBooks");
    }
}
